# 数据库管理页面重构文档

## 1. 项目概述

### 1.1 背景
基于现有的账户管理页面和账户统计页面的成功实现，需要新增数据库管理页面和数据库统计页面。这两个页面将复用现有的设计模式、样式和交互逻辑，但专注于数据库层面的管理和统计功能。

### 1.2 目标
1. **数据库管理页面** - 提供数据库实例和数据库的统一管理界面
2. **数据库统计页面** - 展示数据库相关的统计信息和分析报告
3. **保持一致性** - 与现有账户管理页面保持相同的用户体验和视觉风格

### 1.3 预期收益
- 提供统一的数据库管理入口
- 增强数据库监控和分析能力
- 提升用户操作效率
- 完善系统功能覆盖

### 5.1 数据库统计服务
**文件路径**: `app/services/database_statistics_service.py`

```python
"""数据库统计服务"""

from typing import Dict, Any, List
from sqlalchemy import func, desc
from app import db
from app.models.instance_database import InstanceDatabase
from app.models.database_size_aggregation import DatabaseSizeAggregation
from app.models.instance import Instance

def build_aggregated_statistics() -> Dict[str, Any]:
    """构建聚合统计数据"""
    
    # 总体统计
    total_databases = InstanceDatabase.query.filter_by(is_active=True).count()
    active_databases = InstanceDatabase.query.filter_by(is_active=True).count()
    total_instances = Instance.query.filter_by(is_active=True).count()
    
    # 计算总大小
    total_size_query = db.session.query(
        func.sum(DatabaseSizeAggregation.avg_size_mb)
    ).join(InstanceDatabase).filter(InstanceDatabase.is_active == True)
    total_size_mb = total_size_query.scalar() or 0
    total_size_gb = round(total_size_mb / 1024, 2)
    
    # 按数据库类型统计
    db_type_stats = fetch_db_type_stats()
    
    return {
        'total_databases': total_databases,
        'active_databases': active_databases,
        'total_instances': total_instances,
        'total_size_gb': total_size_gb,
        'db_type_stats': db_type_stats,
    }

def fetch_db_type_stats() -> Dict[str, Dict[str, Any]]:
    """获取按数据库类型的统计"""
    
    stats = {}
    db_types = ['mysql', 'postgresql', 'oracle', 'sqlserver']
    
    for db_type in db_types:
        count = (InstanceDatabase.query
                .join(Instance)
                .filter(Instance.db_type == db_type, InstanceDatabase.is_active == True)
                .count())
        
        size_query = (db.session.query(func.sum(DatabaseSizeAggregation.avg_size_mb))
                     .join(InstanceDatabase)
                     .join(Instance)
                     .filter(Instance.db_type == db_type, InstanceDatabase.is_active == True))
        size_mb = size_query.scalar() or 0
        size_gb = round(size_mb / 1024, 2)
        
        stats[db_type] = {
            'count': count,
            'size_mb': size_mb,
            'size_gb': size_gb,
        }
    
    return stats

def fetch_size_trends(days: int = 30) -> List[Dict[str, Any]]:
    """获取数据库大小趋势数据"""
    
    from datetime import date, timedelta
    end_date = date.today()
    start_date = end_date - timedelta(days=days)
    
    trends = (DatabaseSizeAggregation.query
             .filter(DatabaseSizeAggregation.period_start >= start_date)
             .filter(DatabaseSizeAggregation.period_end <= end_date)
             .order_by(DatabaseSizeAggregation.period_start)
             .all())
    
    return [trend.to_dict() for trend in trends]
```

```python
"""数据库管理服务"""

from typing import Dict, Any, Optional
from app.models.instance_database import InstanceDatabase
from app.models.instance import Instance
from app.services.connection_test_service import connection_test_service

def get_database_details(database_id: int) -> Dict[str, Any]:
    """获取数据库详情"""
    
    database = InstanceDatabase.query.get_or_404(database_id)
    instance = database.instance
    
    details = {
        'id': database.id,
        'database_name': database.database_name,
        'is_active': database.is_active,
        'first_seen_date': database.first_seen_date,
        'last_seen_date': database.last_seen_date,
        'instance': {
            'id': instance.id,
            'name': instance.name,
            'host': instance.host,
            'port': instance.port,
            'db_type': instance.db_type,
        } if instance else None,
    }
    
    return details

def test_database_connection(database_id: int) -> Dict[str, Any]:
    """测试数据库连接"""
    
    database = InstanceDatabase.query.get_or_404(database_id)
    instance = database.instance
    
    if not instance:
        return {
            'success': False,
            'error': '数据库实例不存在'
        }
    
    # 使用现有的连接测试服务
    result = connection_test_service.test_connection(instance)
    
    return result
```
